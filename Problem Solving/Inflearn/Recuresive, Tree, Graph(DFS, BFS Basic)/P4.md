package P7;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class P4 {
    // static int[] fibo;
    public int solution(int n) {
        // 피보나치의 항의 값을 구하기 위한 메서드임을 기억하자
//      if(fibo[n]!=0) return fibo[n];
        // 처음에 0으로 초기화되는데, 0이 아니라는 것은 값이 이미 구해져 있단 얘기임
        if(n==1 || n==2) return 1;
//        if(n==1) fino[n] = 1;
//        if(n==2) fino[n] = 1;
        else return solution(n - 1) + solution(n - 2);
//        else return fibo[n] = solution(n - 1) + solution(n - 2);
    }
    public static void main(String[] args) {
        FastReader scan = new FastReader();
        P4 p = new P4();
        int n = scan.nextInt();
//        fibo = new int[n+1]
//        피보나치의 값을 저장하기 위한 배열, 0번째 index는 사용하지 않는다.
        for (int i = 1; i <= n; i++) {
            System.out.print(p.solution(i)+ " ");
        }
//      p.solution(n);
//      for(int i=1; i<=n; i++) System.out.println(fibo[i] + " ");
//         재귀함수는 한 번만 실행하고, 배열에 저장되어 있는 값을 출력한다
//         재귀를 한 번만 돌려도 나머지 값을 다 채울 수 있기 때문에!
    }
    public static class FastReader{
            BufferedReader br;
            StringTokenizer st;

            FastReader(){
                br = new BufferedReader(new InputStreamReader(System.in));
            }

            String next(){
                while (st == null || !st.hasMoreElements()) {
                    try {
                        st = new StringTokenizer(br.readLine());
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
                return st.nextToken();
            }
            int nextInt(){
                return Integer.parseInt(next());
            }
        }
}
